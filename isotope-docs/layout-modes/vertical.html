Options

horizontalAlignment

Aligns items horizontally. Set as a decimal 0 to 1.

/* default
vertical: {
  // align to origin edge
  horizontalAlignment: 0
} */

vertical: {
  // align to center
  horizontalAlignment: 0.5
}

vertical: {
  // align to opposite edge
  horizontalAlignment: 1
}

layoutMode: 'vertical',
vertical: {
  horizontalAlignment: 0.5,
}
